\section{Gas Turbine Digital Twin (Engineering Thesis)}
\begin{frame}
\frametitle{Engineering Thesis: Gas Turbine Digital Twin}
\textbf{Context:} Real-time monitoring of GE LM2500 gas turbines for Sonatrach. \\
\textbf{Challenge:} Hardware Latency Paradox (Cloud AI is too slow for emergency shutdowns, edge hardware cannot process massive models).\\
\textbf{Solution:} A "Dual-Track" 3-Tier Pipeline decoupling the logic.
\vspace{0.3cm}
\begin{itemize}
    \item \textbf{TCNEdge (Temporal Convolutional Network):} 1D CNNs optimized for the Edge. Acts as an immediate safety trigger, bypassing complex attention mechanisms for pure speed (0.049 ms inference).
    \item \textbf{MambaTS (Selective State Space Model):} A near-edge sequence model providing linear $O(N)$ complexity for infinite-horizon memory, enhanced with a "missingness-aware gate" for sensor faults.
    \item \textbf{ST-KDFormer \& KAN-PCNN (Cloud):} Spatio-Temporal Knowledge-Distilled Transformers map complex interactions, while Kolmogorov-Arnold Networks act as Physics-Constrained NNs to obey thermodynamics.
\end{itemize}
\end{frame}

\section{Medical Digital Twin (Master's Thesis)}
\begin{frame}
\frametitle{Master's Thesis: Cardiovascular Digital Twin}
\textbf{Context:} Predicting sudden drops in Mean Blood Pressure (MBP) in Intensive Care. \\
\textbf{Challenge:} Models overfit to time-series noise and ignore fundamental fluid dynamics.\\
\textbf{Data:} 100Hz VitalDB dataset with strict 70/15/15 patient-level split.
\vspace{0.3cm}
\begin{itemize}
    \item \textbf{PI\_DeepONet (Physics-Informed Deep Operator Network):} A dual-network architecture (Branch Net for patient anatomy, Trunk Net for time). It learns mathematical operators (PDEs) rather than just functions.
    \item \textbf{0D Windkessel Model:} Acts as the physics engine. PyTorch Autograd calculates derivatives to penalize predictions violating physical vascular resistance or arterial compliance.
    \item \textbf{Medical\_MambaTS:} Deployed at the bedside for continuous state-space monitoring without memory bottlenecks.
\end{itemize}
\end{frame}

\section{Enterprise AI Projects}
\begin{frame}
\frametitle{Project: Aura — Autonomous Trading System}
\textbf{Architecture:} Bridging Sequence Learning and Reinforcement Learning
\vspace{0.3cm}
\begin{itemize}
    \item \textbf{LSTM (Long Short-Term Memory):} A specialized Recurrent Neural Network used in the Prediction Layer. It processes historical market data to capture long-term temporal dependencies and predict short-term price movements.
    \item \textbf{DQN (Deep Q-Network):} Used in the Action Layer. A Reinforcement Learning algorithm that combines Q-Learning with deep neural networks. It takes LSTM outputs as states to optimize long-term trading reward policies.
    \item \textbf{Infrastructure:} Highly scalable \textbf{FastAPI} backend and a reactive \textbf{Next.js} dashboard receiving real-time agent telemetry via WebSockets.
\end{itemize}
\end{frame}
